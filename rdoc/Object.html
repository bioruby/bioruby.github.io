def initialize_single_source(root)
  inf = 1 / 0.0                               # inf.infinite? -> true

  distance = {}
  predecessor = {}

  @graph.each_key do |k|
    distance[k] = inf
    predecessor[k] = nil
  end
  distance[root] = 0
  return distance, predecessor
end
