def Util::array_to_string( ary )
  str = String.new
  ary.each do | e |
    str << e << " "
  end
  str.chomp!( " " )
  str  
end

def Util::to_nexus_helper( block, lines )
  str = String.new
  str << BEGIN_BLOCK << " " << block << END_OF_LINE
  lines.each do | line |
    if ( line != nil )
      str << INDENTENTION << line << END_OF_LINE
    end
  end # do
  str << END_BLOCK << END_OF_LINE
  str
end
