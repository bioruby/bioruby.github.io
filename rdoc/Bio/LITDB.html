def reference
  hash = Hash.new('') 

  hash['authors'] = author.split(/;/).map {|x| x.sub(/,/, ', ')}
  hash['title']   = title 
  hash['journal'] = journal.gsub(/\./, '. ').strip

  vol = volume.split(/,\s+/)
  if vol.size > 1
    hash['volume'] = vol.shift.sub(/Vol\./, '')

    hash['year'] = vol.pop.split(' ')
    hash['issue'] = vol.shift.sub(/No\./, '') unless vol.empty?
  end

  return Reference.new(hash)
