def to_xml
  distr = LibXML::XML::Node.new('distribution')
  PhyloXML::Writer.generate_xml(distr, self, [
      [:simple, 'desc', @desc],
      [:objarr, 'point', 'points'],
      [:objarr, 'polygon', 'polygons']])
  return distr
end
