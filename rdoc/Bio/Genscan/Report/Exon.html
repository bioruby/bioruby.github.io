def self.parser(line)
  e = line.strip.split(/ +/)
  case line
  when /PlyA/, /Prom/
    e[12] = e[6].clone
    e[11] = 0
    [6,7,8,9,10].each {|i| e[i] = nil }
  end
  self.new(e[0], e[1], e[2], e[3], e[4], e[5], e[6], 
           e[7], e[8], e[9], e[10], e[11], e[12])
end
