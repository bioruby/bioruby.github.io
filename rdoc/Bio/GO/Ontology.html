39 def self.parse_goids(line)
40   goids = []
41   loop {
42     if /^ *[$%<]\S.+?;/ =~ line
43       endpoint = line.index(';') + 1
44       line = line[endpoint..line.size]
45     elsif /^,* GO:(\d{7}),*/ =~ line
46       goids << $1.clone
47       endpoint = line.index(goids.last) + goids.last.size
48       line = line[endpoint..line.size]
49     else
50       break
51     end
52   }
53   return goids
54 end
