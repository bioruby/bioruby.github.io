def initialize(dbname, nsclass, arg)
  @dbname = dbname
  @nsclass = nsclass
  if arg.is_a?(String) then
    a = arg.split("\t")
  else
    a = arg
  end
  a.each do |x|
    self[x] = @nsclass.new(@dbname, x)
  end
  self
end

def search(key)
  r = []
  values.each do |ns|
    r.concat ns.search(key)
  end
  r.sort!
  r.uniq!
  r
end

def search_names(key, *names)
  r = []
  names.each do |x|
    ns = self[x]
    raise "undefined namespace #{x.inspect}" unless ns
    r.concat ns.search(key)
  end
  r
end
