def to_re(seq)
  replace = {
    'B' => '[DNB]',
    'Z' => '[EQZ]',
    'J' => '[ILJ]',
    'X' => '[ACDEFGHIKLMNPQRSTVWYUOX]',
  }
  replace.default = '.'

  str = seq.to_s.upcase
  str.gsub!(/[^ACDEFGHIKLMNPQRSTVWYUO]/) { |aa|
    replace[aa]
  }
  Regexp.new(str)
end

def weight(x = nil)
  return WEIGHT unless x

  if x.length > 1
    total = 0.0
    x.each_byte do |byte|
      aa = byte.chr.upcase
      if WEIGHT[aa]
        total += WEIGHT[aa]
      else
        raise "Error: invalid amino acid '#{aa}'"
      end
    end
    total -= NucleicAcid.weight[:water] * (x.length - 1)
  else
    WEIGHT[x]
  end
end
