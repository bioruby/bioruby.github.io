def dblinks_as_hash
  unless defined? @dblinks_as_hash
    hash = {}
    dblinks_as_strings.each do |line|
      db, ids = line.split(/\:\s*/, 2)
      list = ids.split(/\s+/)
      hash[db] = list
    end
    @dblinks_as_hash = hash
  end
  @dblinks_as_hash
end
