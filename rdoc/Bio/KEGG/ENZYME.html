    kegg_reactions() click to toggle source 
    # File lib/bio/db/kegg/enzyme.rb, line 91
def kegg_reactions
  reac = all_reac
  if reac[/\(other\)/]
    reac.sub(/.*\(other\)\s*/, '').split(/\s*;\s*/)
  else
    []
  end
end

def structures

    @data['STRUCTURES'] = fetch('STRUCTURES').sub(/(PDB: )*/,'').split(/\s+/)
  end
  @data['STRUCTURES']
