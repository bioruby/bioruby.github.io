def rel_pathways_as_hash
  unless defined? @rel_pathways_as_hash then
    hash = {}
    rel_pathways_as_strings.each do |line|
      entry_id, name = line.split(/\s+/, 2)
      hash[entry_id] = name
    end
    @rel_pathways_as_hash = hash
  end
  @rel_pathways_as_hash
end
