def report
  report =  "### siRNA\n"
  report << 'Start: ' + @start.to_s + "\n"
  report << 'Stop:  ' + @stop.to_s  + "\n"
  report << 'Rule:  ' + @rule.to_s  + "\n"
  report << 'GC %:  ' + @gc_percent.to_s  + "\n"
  report << 'Target:    '        + @target.upcase + "\n"
  report << 'Sense:     ' + '  ' + @sense.upcase  + "\n"
  report << 'Antisense: '        + @antisense.reverse.upcase + "\n"
end
