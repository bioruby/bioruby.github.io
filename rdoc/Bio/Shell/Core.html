def ask_yes_or_no(message)
  loop do
    STDERR.print "#{message}"
    answer = gets
    if answer.nil?
      # readline support might be broken
      return false
    elsif /^\s*[Nn]/.match(answer)
      return false
    elsif /^\s*[Yy]/.match(answer)
      return true
    else
      # loop
    end
  end
end
