    class Bio::Blast::WU::Report::Hit

    Because it inherits Bio::Blast::Default::Report::Hit, please also refer Bio::Blast::Default::Report::Hit.

def initialize(data)
  @f0hitname = data.shift
  @hsps = []
  while r = data.first
    if r =~ /^\s*(?:Plus|Minus) +Strand +HSPs\:/ then
      data.shift
      r = data.first
    end
    if /\A\s+Score/ =~ r then
      @hsps << HSP.new(data)
    else
      break
    end
  end
  @again = false
end
