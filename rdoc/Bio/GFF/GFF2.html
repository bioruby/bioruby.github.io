def parse(str)
  # parses GFF lines
  str.each_line do |line|
    if /^\#\#([^\s]+)/ =~ line then
      parse_metadata($1, line)
    else
      @records << GFF2::Record.new(line)
    end
  end
  self
end

def to_s
  ver = @gff_version || VERSION.to_s
  ver = ver.gsub(/[\r\n]+/, ' ')
  ([ "##gff-version #{ver}\n" ] +
   @metadata.collect { |m| m.to_s } +
   @records.collect{ |r| r.to_s }).join('')
end
