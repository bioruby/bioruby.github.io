def journal(ids, format = "full", hash = {})
  case format
  when "xml"
    format = "full"
    mode = "xml"
  else
    mode = "text"
  end
  opts = { "db" => "journals", "rettype" => format, "retmode" => mode }
  opts.update(hash)
  Bio::NCBI::REST.efetch(ids, opts)
end
