def get_all_by_qualifier(qstr)
  unless defined?(@hash)
    @hash = {}
  end
  unless @hash.member?(qstr) then
    @hash[qstr] = self.find_all do |x|
      x.qualifier == qstr
    end
  end
  @hash[qstr]
end

def to_a
  unless defined?(@a)
    @a = @elem.get_elements('annotation')
    @a.collect! { |e| MaXML::Annotation.new(e) }
  end
  @a
end
